```css

/* Small devices (640px+) */
@media (min-width: 640px) {
    .container { padding: 0 1.5rem; }
    .form-row { grid-template-columns: repeat(2, 1fr); }
}

/* Large devices (1024px+) */
@media (min-width: 1024px) {
    .container { padding: 0 2rem; }

}
```
